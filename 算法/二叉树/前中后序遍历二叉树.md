## 前序遍历二叉树

```javascript
let traversal = function (root) {
  let arr = [];
  function traverse(root) {
    if (root) {
      arr.push(root.val);
      if (root.left) traverse(root.left);
      if (root.right) traverse(root.right);
    }
  }
  traverse(root);
  return arr;
};
```

## 后序遍历二叉树

```javascript
let traversal = function (root) {
  let arr = [];
  function traverse(root) {
    if (root) {
      if (root.left) traversal(root.left);
      if (root.right) traverse(root.right);
      arr.push(root.val);
    }
  }
  traverse(root);
  return arr;
};
```
